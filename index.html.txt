// BOOT SEQUENCE
const boot = document.getElementById("boot");
const bootLines = [
  "Initializing kernel...",
  "Loading security modules...",
  "Connecting to secure node...",
  "Authenticating user...",
  "Access Granted."
];

let b = 0;
function bootSeq() {
  if (b < bootLines.length) {
    boot.innerHTML += bootLines[b] + "<br>";
    b++;
    setTimeout(bootSeq, 500);
  } else {
    setTimeout(() => boot.style.display = "none", 1000);
  }
}
bootSeq();

// MATRIX
const matrix = document.getElementById("matrix");
const ctxM = matrix.getContext("2d");

matrix.width = window.innerWidth;
matrix.height = window.innerHeight;

const letters = "01";
const fontSize = 14;
const columns = matrix.width / fontSize;
const drops = Array.from({length: columns}).fill(1);

function drawMatrix() {
  ctxM.fillStyle = "rgba(0,0,0,0.05)";
  ctxM.fillRect(0, 0, matrix.width, matrix.height);

  ctxM.fillStyle = "#00ff9f";
  ctxM.font = fontSize + "px monospace";

  drops.forEach((y, i) => {
    const text = letters[Math.floor(Math.random()*letters.length)];
    ctxM.fillText(text, i * fontSize, y * fontSize);

    if (y * fontSize > matrix.height && Math.random() > 0.975) {
      drops[i] = 0;
    }
    drops[i]++;
  });
}
setInterval(drawMatrix, 33);

// PARTICLE NETWORK
const net = document.getElementById("network");
const ctxN = net.getContext("2d");

net.width = window.innerWidth;
net.height = window.innerHeight;

let mouse = {x: null, y: null};

window.addEventListener("mousemove", e => {
  mouse.x = e.x;
  mouse.y = e.y;
});

let particles = [];
for (let i = 0; i < 80; i++) {
  particles.push({
    x: Math.random()*net.width,
    y: Math.random()*net.height,
    dx: Math.random()-0.5,
    dy: Math.random()-0.5
  });
}

function drawNetwork() {
  ctxN.clearRect(0,0,net.width,net.height);

  particles.forEach(p => {
    p.x += p.dx;
    p.y += p.dy;

    if (p.x < 0 || p.x > net.width) p.dx *= -1;
    if (p.y < 0 || p.y > net.height) p.dy *= -1;

    particles.forEach(p2 => {
      let dx = p.x - p2.x;
      let dy = p.y - p2.y;
      let dist = Math.sqrt(dx*dx + dy*dy);

      if (dist < 100) {
        ctxN.strokeStyle = "#00ff9f";
        ctxN.beginPath();
        ctxN.moveTo(p.x, p.y);
        ctxN.lineTo(p2.x, p2.y);
        ctxN.stroke();
      }
    });

    // mouse interaction
    let dx = p.x - mouse.x;
    let dy = p.y - mouse.y;
    let dist = Math.sqrt(dx*dx + dy*dy);

    if (dist < 120) {
      ctxN.strokeStyle = "red";
      ctxN.beginPath();
      ctxN.moveTo(p.x, p.y);
      ctxN.lineTo(mouse.x, mouse.y);
      ctxN.stroke();
    }
  });

  requestAnimationFrame(drawNetwork);
}
drawNetwork();

// TERMINAL
const terminal = document.getElementById("terminal");

function print(text) {
  const line = document.createElement("div");
  line.className = "line";
  line.textContent = text;
  terminal.appendChild(line);
}

function inputLine() {
  const line = document.createElement("div");
  const input = document.createElement("input");

  line.append("> ");
  line.appendChild(input);
  terminal.appendChild(line);

  input.focus();

  input.addEventListener("keydown", e => {
    if (e.key === "Enter") {
      handle(input.value);
      input.disabled = true;
      inputLine();
    }
  });
}

function handle(cmd) {
  if (cmd === "help") print("help, whoami, skills, clear");
  else if (cmd === "whoami") print("Aditya Kumar - Cybersecurity Analyst");
  else if (cmd === "skills") print("DFIR | SOC | Blue Team");
  else if (cmd === "clear") terminal.innerHTML = "";
  else print("Unknown command");
}

print("System ready...");
inputLine();

// SCAN
function scan() {
  print("Scanning system...");
  let p = 0;
  let int = setInterval(() => {
    p += 10;
    print("Scan " + p + "%");
    if (p >= 100) {
      clearInterval(int);
      print("System Secure.");
    }
  }, 300);
}
